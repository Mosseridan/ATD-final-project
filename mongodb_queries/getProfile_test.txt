db.profiles.aggregate([
    {  $match : {
        "username": "liad"
    }},

    { $group: {
        "_id": "$follower",
        "following": {
            $push: "$following" 
        }
    }},
])
     {  $lookup: 
     {
       "from": "followers",
       "localField": "_id",
       "foreignField": "following",
       "as": "followings"
     }},
     {
         $unwind: "$followings"
     },

     { $group: {
        "_id": { "username": "$_id", "following" : "$following" }, 
        
        "followers": {
            $push: "$followings.follower" 
        }
        
    }},
    {
        $project : {
            "_id": 0,
            "username": "$_id.username",
            "following": "$_id.following",
            "followers" : 1,
            
        }
    }
    ])
